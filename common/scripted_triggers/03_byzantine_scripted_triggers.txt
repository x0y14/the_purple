tp_character_any_involvement_byzantine_struggle_trigger = {
	has_fp2_dlc_trigger = yes
	OR = {
		tp_character_involved_in_byzantine_trigger = yes
		tp_character_interloper_in_byzantine_trigger = yes
	}
}

tp_character_involved_in_byzantine_trigger = {
	any_character_struggle = {
		involvement = involved
		is_struggle_type = byzantine_struggle
	}
}

tp_character_interloper_in_byzantine_trigger = {
	any_character_struggle = {
		involvement = interloper
		is_struggle_type = byzantine_struggle
	}
}

tp_is_pregnant_of_purple_trigger = {
	AND = {
		is_female = yes
		is_pregnant = yes
		OR = {
			has_title = title:e_byzantium
			any_spouse = {
				has_title = title:e_byzantium
			}
		}
	}
}
