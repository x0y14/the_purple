}


catalyst_yearly_emperor_leaning_gain = {
	value = 0
	# The learning capacity of the emperor is the base value.
	add = title:e_byzantium.holder.learning
	
	# Learning trait
	if = {
		limit = {
			title:e_byzantium.holder = {
				has_trait = education_learning_1
			}
		}
		multiply = 1.1
	}
	if = {
		limit = {
			title:e_byzantium.holder = {
				has_trait = education_learning_2
			}
		}
		multiply = 1.2
	}
	if = {
		limit = {
			title:e_byzantium.holder = {
				has_trait = education_learning_3
			}
		}
		multiply = 1.3
	}
	if = {
		limit = {
			title:e_byzantium.holder = {
				has_trait = education_learning_4
			}
		}
		multiply = 1.4
	}
	if = {
		limit = {
			title:e_byzantium.holder = {
				has_trait = education_learning_5
			}
		}
		multiply = 1.5
	}

	# Scholarly trait
	if = {
		limit = {
			title:e_byzantium.holder = {
				has_trait = scholarly_court_1
			}
		}
		multiply = 1.2
	}
	if = {
		limit = {
			title:e_byzantium.holder = {
				has_trait = scholarly_court_2
			}
		}
		multiply = 1.4
	}

	# Intellect trait
	if = {
		limit = {
			title:e_byzantium.holder = {
				has_trait = intellect_good_1
			}
		}
		multiply = 1.25
	}
	if = {
		limit = {
			title:e_byzantium.holder = {
				has_trait = intellect_good_2
			}
		}
		multiply = 1.5
	}
	if = {
		limit = {
			title:e_byzantium.holder = {
				has_trait = intellect_good_3
			}
		}
		multiply = 2.0
	}

	# Court
	if = {
		limit = {
			title:e_byzantium.holder = {
				has_court_type = court_scholarly
			}
		}
		multiply = 2.0
	}

	divide = 3
}
